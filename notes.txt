# Create a network
docker network create pg-network

# Run postgres
docker run -it \
  -e POSTGRES_USER="root" \
  -e POSTGRES_PASSWORD="root" \
  -e POSTGRES_DB="ny_taxi" \
  -v $(pwd)/ny_taxi_postgres_data:/var/lib/postgresql/data \
  -p 5432:5432 \
  --network=pg-network \
  --name pg-database \
  postgres:13

# Run pgadmin
docker run -it \
  -e PGADMIN_DEFAULT_EMAIL="admin@admin.com" \
  -e PGADMIN_DEFAULT_PASSWORD="root" \
  -p 8080:80 \
  --network=pg-network \
  --name pgadmin-2 \
  dpage/pgadmin4

# Open in browser the following localhost:8080 and connect to pgadmin with the above credentials
# Register a new server using the pg-database in host name/address

# Ingest data
URL="https://github.com/DataTalksClub/nyc-tlc-data/releases/download/green/green_tripdata_2019-09.csv.gz"

python ingest_data_hw.py \
  --user=root \
  --password=root \
  --host=localhost \
  --port=5432 \
  --db=ny_taxi \
  --table_name=green_taxi_trips \
  --url=${URL}



SELECT dro."Zone", SUM(tt."tip_amount")
FROM public.green_taxi_trips AS tt
INNER JOIN public.zones AS pu ON tt."PULocationID" = pu."LocationID"
INNER JOIN public.zones AS dro ON tt."DOLocationID" = dro."LocationID"
WHERE pu."Zone" <> 'Astoria'
  AND CAST(lpep_pickup_datetime AS DATE) BETWEEN '2019-09-01' AND '2019-09-30'
GROUP BY dro."Zone"
ORDER BY 2 DESC
